# 七.泛型

# 1.指定函数参数类型

  • 单个泛型
const getArray = <T>(times: number, val: T): T[] => {
  let result: T[] = []
  for (let i = 0; i < times; i++) {
    result.push(val)
  }
  return result
}
getArray(3, 3) // 3 => T => number

# 5.类中的泛型

  • 创建实例时提供类型
class MyArray<T> {
  // T=> number
  arr: T[] = []
  add(num: T) {
    this.arr.push(num)
  }
  getMaxNum(): T {
    let arr = this.arr
    let max = arr[0]
    for (let i = 1; i < arr.length; i++) {
      let current = arr[i]
      current > max ? (max = current) : null
    }
  }
}
let myArr = new MyArray<number>()
myArr.add(3)
myArr.add(1)
myArr.add(2)
console.log(myArr.getMaxNum())

# 6.泛型约束

  • 泛型必须包含某些属性
interface IWithLength {
  length: number
}
function getLen<T extends IWithLength>(val: T) {
  return val.length
}
getLen("hello")
